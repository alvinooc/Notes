1. 在 html 中引入 iframe

   ```html
   <iframe :src="iframeUrl"
   				id="frame"
   				width="100%"
   				height="100%"
   				class="system-monitor__iframe" />
   ```

2. 在 data 中设置 iframeUrl 的值

   ```js
   data() {
   	return {
   		iframeUrl: ''
   	}
   }
   ```

3. 在 mounted 中初始化 iframeUrl

   ```js
   mounted() {
   	this.iframeUrl = '所需地址';
   }
   ```

4. 在 watch 中监听 iframeUrl（通过地址变化监听跳转）

   ```js
   watch: {
   	iframeUrl() {
   		const iframe = document.getElementById('frame');
   		iframe.onLoad = () => {
   			console.log('页面跳转');
   		}
   	}
   }
   ```
